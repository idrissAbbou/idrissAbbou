there is 4 types of rederiction operations in the shell
The > symbol is used for output (STDOUT) redirection.
The >> is to append data, for example we can append a new line to the test.txt like this:

first lets create our algorithme to extract all data
copy or create ur own. 

include < iostream >
using namespace std;
int main (int argc, char** argv) {
    //variables declaration
    size_t pos = 0;
    string del = ",";
    string token;
    //I used vectors to stock the data ( vectors are like arrays but they are dynamic u can add to them automaticly)
    vector < string > tokens;
    //this is a vector of vector
    vector < vector < string > > data;
    for (string line; getline(cin, line);) {
        while ((pos = line.find(del)) != std::string::npos) {
            tokens.push_back(line.substr(0, pos));
            line.erase(0, pos + del.length());
        }
        tokens.push_back(line.substr(0, pos));
        data.push_back(tokens);
        tokens.clear();
    }
    //printing the data on the stdout 
    for (size_t i = 0; i < data.size(); i++) { 
        cout << data[i][1] << endl; 
    }
    //in c or cpp we should always return 0 from the main this allow us to know from the terminal if the app exec correctly
    return 0;
}
